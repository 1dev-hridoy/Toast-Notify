Toast Notify

Simple Toast Notifications

Lightweight, customizable toast notifications with zero dependencies

Installation

<!-- Add to your HTML head -->
<script src="https://cdn.tailwindcss.com"></script>
<link rel="stylesheet" href="https://cdn.jsdelivr.net/gh/1dev-hridoy/toast-notify@main/dist/toast-notify.min.css">
<script src="https://cdn.jsdelivr.net/gh/1dev-hridoy/toast-notify@main/dist/toast-notify.min.js"></script>

Usage

// Initialize
const toast = new ToastNotify({
    position: 'top-right', // top-right, top-left, bottom-right, bottom-left
    duration: 5000 // Duration in milliseconds
});

// Show notifications
toast.show('success', 'Success message');
toast.show('warning', 'Warning message');
toast.show('error', 'Error message');
