Toast Notify

Simple Toast Notifications

Just 3 lines of code to get started!

Quick Start

<!-- 1. Add these to your HTML head -->
<script src="https://cdn.tailwindcss.com"></script>
<link rel="stylesheet" href="https://cdn.jsdelivr.net/gh/1dev-hridoy/toast-notify@main/dist/toast-notify.min.css">
<script src="https://cdn.jsdelivr.net/gh/1dev-hridoy/toast-notify@main/dist/toast-notify.min.js"></script>

<!-- 2. Initialize toast -->
<script>
    const toast = new ToastNotify();
</script>

<!-- 3. Show notifications -->
toast.show('success', 'Your message here');
